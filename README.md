# 环境
* nodejs > 6.10.0
* npm > 3.3
# 开发环境
* npm install -g webpack
* npm install -g babel-cli
* npm install -g grunt-cli
# 开发包
* npm install
# 启动
* npm start
# 打包
* grunt


# 部署
## 在项目根目录下
* 在.bashrc中 export NODE_ENV=production
* npm install -g pm2
* npm install

# 启动
* pm2 start publish/server.js
# 重新加载
* pm2 reload all
